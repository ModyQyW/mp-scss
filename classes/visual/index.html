// 通用变量

  "-full": 100%,
  "-half": 50%,

  "-sm": (

  ) * $scale + $unit,
  "-md": $border-radius-base * $scale + $unit,

  "-2xl": (
    $border-radius-base + $border-radius-diff * 4

  "-3xl": (
    $border-radius-base + $border-radius-diff * 6
